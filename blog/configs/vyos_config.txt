# VyOS 路由器配置示例
# 这是 VPC 虚拟路由器的完整配置

# ============================================
# 1. 基础接口配置
# ============================================

# 子网 A 的接口（VLAN 100）
set interfaces ethernet eth0 vlan 100 address 10.0.1.1/24
set interfaces ethernet eth0 vlan 100 description "VPC-1 Subnet-A"

# 子网 B 的接口（VLAN 101）
set interfaces ethernet eth0 vlan 101 address 10.0.2.1/24
set interfaces ethernet eth0 vlan 101 description "VPC-1 Subnet-B"

# 外网出口接口
set interfaces ethernet eth1 address 192.168.100.1/24
set interfaces ethernet eth1 description "Uplink-WAN"

# 环回地址
set interfaces loopback lo address 127.0.0.1/8

# ============================================
# 2. 静态路由配置
# ============================================

# 默认路由指向上一级网关
set protocols static route 0.0.0.0/0 next-hop 192.168.100.254

# 跨 VPC 路由（可选）
set protocols static route 10.0.0.0/8 next-hop 192.168.50.1

# ============================================
# 3. NAT 配置（关键！）
# ============================================

# 源 NAT（虚机访问外网）
set nat source rule 100 outbound-interface eth1
set nat source rule 100 source address 10.0.0.0/16
set nat source rule 100 translation address 61.152.10.20   # 外网 IP (EIP)

# 目标 NAT（外网访问虚机服务）
# 例如：将外网的 61.152.10.20:80 转发到内网虚机 10.0.1.100:80
set nat destination rule 10 inbound-interface eth1
set nat destination rule 10 destination address 61.152.10.20
set nat destination rule 10 destination port 80
set nat destination rule 10 translation address 10.0.1.100
set nat destination rule 10 translation port 80
set nat destination rule 10 protocol tcp

# ============================================
# 4. 防火墙/安全组配置
# ============================================

# 入站规则链（应用到 eth0 VLAN 接口）
set firewall name INBOUND_FILTER rule 100 action accept
set firewall name INBOUND_FILTER rule 100 source address 10.0.1.0/24
set firewall name INBOUND_FILTER rule 100 destination address 10.0.2.0/24
set firewall name INBOUND_FILTER rule 100 protocol all

# HTTP 允许
set firewall name INBOUND_FILTER rule 200 action accept
set firewall name INBOUND_FILTER rule 200 protocol tcp
set firewall name INBOUND_FILTER rule 200 destination port 80
set firewall name INBOUND_FILTER rule 200 source address 0.0.0.0/0

# HTTPS 允许
set firewall name INBOUND_FILTER rule 300 action accept
set firewall name INBOUND_FILTER rule 300 protocol tcp
set firewall name INBOUND_FILTER rule 300 destination port 443
set firewall name INBOUND_FILTER rule 300 source address 0.0.0.0/0

# SSH 限制（仅 VPC 内）
set firewall name INBOUND_FILTER rule 400 action accept
set firewall name INBOUND_FILTER rule 400 protocol tcp
set firewall name INBOUND_FILTER rule 400 destination port 22
set firewall name INBOUND_FILTER rule 400 source address 10.0.0.0/16

# 默认拒绝
set firewall name INBOUND_FILTER rule 9999 action drop
set firewall name INBOUND_FILTER rule 9999 protocol all

# 应用防火墙到接口
set interfaces ethernet eth0 firewall in name INBOUND_FILTER

# ============================================
# 5. Keepalived 集成
# ============================================

# 虚拟 IP（由 Keepalived 管理）
set interfaces loopback lo address 10.0.1.1/32
set interfaces loopback lo address 10.0.2.1/32

# 高可用���控接口
set service ssh listen-address 0.0.0.0

# ============================================
# 6. OSPF 动态路由（可选）
# ============================================

set protocols ospf area 0.0.0.0
set protocols ospf area 0.0.0.0 network 10.0.0.0/16
set protocols ospf area 0.0.0.0 network 192.168.100.0/24
set protocols ospf parameters abr-type cisco

# ============================================
# 7. QoS 配置（可选）
# ============================================

# 限制虚机出站带宽为 1Gbps
set traffic-policy shaper VM_SHAPER bandwidth 1000mbit
set traffic-policy shaper VM_SHAPER class 1 bandwidth 1000mbit

set interfaces ethernet eth0 traffic-policy out VM_SHAPER

# ============================================
# 8. 系统配置
# ============================================

# 启用 IP 转发
set system sysctl parameter net.ipv4.ip_forward value 1

# 禁用 rp_filter（允许非对称路由）
set system sysctl parameter net.ipv4.conf.all.rp_filter value 0
set system sysctl parameter net.ipv4.conf.default.rp_filter value 0

# 增加 conntrack 表大小
set system sysctl parameter net.nf_conntrack_max value 1000000

# ============================================
# 9. 系统管理
# ============================================

# 时间同步
set system ntp server 192.168.1.1

# DNS 配置
set system name-server 8.8.8.8
set system name-server 8.8.4.4

# 主机名
set system host-name vpc-router-1

# ============================================
# 配置保存与提交
# ============================================

# 提交配置
commit

# 保存配置到文件
save
